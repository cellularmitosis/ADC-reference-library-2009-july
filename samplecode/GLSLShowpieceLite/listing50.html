static float tex[2][2][2] =
{
  { {0, 0}, {1, 0}},
  { {0, 1}, {1, 1}},
};

// Returns the number of vertices sent to GL.

int DrawTeapot(GLint grid, GLdouble scale, GLenum type)
{
  GLfloat   p[4][4][3];
  GLfloat   q[4][4][3];
  GLfloat   r[4][4][3];
  GLfloat   s[4][4][3];
  
  int  i;
  int  j;
  int  k;
  int  l;
  
  int  numVertices = 0;

  // Make polygons wind CCW by scaling by -1.
  
  scale *= -1;

  glPushAttrib(GL_ENABLE_BIT | GL_EVAL_BIT);
  
    glEnable(GL_AUTO_NORMAL);
    glEnable(GL_NORMALIZE);
    glEnable(GL_MAP2_VERTEX_3);
    glEnable(GL_MAP2_TEXTURE_COORD_2);
    
    glPushMatrix();
    
      glRotatef(90.0, 1.0, 0.0, 0.0);
      glScalef(0.5 * scale, 0.5 * scale, 0.5 * scale);
      glTranslatef(0.0, 0.0, -1.5);

      for (i = 0; i < 10; i++)
      {
        for (j = 0; j < 4; j++)
        {
          for (k = 0; k < 4; k++)
          {
            for (l = 0; l < 3; l++)
            {
              p[j][k][l] = cpdata[patchdata[i][j * 4 + k]][l];
              q[j][k][l] = cpdata[patchdata[i][j * 4 + (3 - k)]][l];
              
              if (l == 1)
              {
                q[j][k][l] *= -1.0;
              } // if
              
              if (i < 6)
              {
                r[j][k][l] = cpdata[patchdata[i][j * 4 + (3 - k)]][l];
                
                if (l == 0)
                {
                  r[j][k][l] *= -1.0;
                } // if
                
                s[j][k][l] = cpdata[patchdata[i][j * 4 + k]][l];
                
                if (l == 0)
                {
                  s[j][k][l] *= -1.0;
                } // if
                
                if (l == 1)
                {
                  s[j][k][l] *= -1.0;
                } // if
              } // if
            } // for l
          } // for k
        } // for j
        
        glMap2f(GL_MAP2_TEXTURE_COORD_2, 0, 1, 2, 2, 1, 0, 4, 2, &tex[0][0][0]);
        glMap2f(GL_MAP2_VERTEX_3, 0, 1, 3, 4, 0, 1, 12, 4, &p[0][0][0]);
        glMapGrid2f(grid, 0.0, 1.0, grid, 0.0, 1.0);
        glEvalMesh2(type, 0, grid, 0, grid);
        
        numVertices += grid * grid;
        
        glMap2f(GL_MAP2_VERTEX_3, 0, 1, 3, 4, 0, 1, 12, 4, &q[0][0][0]);
        glEvalMesh2(type, 0, grid, 0, grid);
        
        numVertices += grid * grid;
        
        if (i < 6)
        {
          glMap2f(GL_MAP2_VERTEX_3, 0, 1, 3, 4, 0, 1, 12, 4, &r[0][0][0]);
          glEvalMesh2(type, 0, grid, 0, grid);
          
          numVertices += grid * grid;
          
          glMap2f(GL_MAP2_VERTEX_3, 0, 1, 3, 4, 0, 1, 12, 4, &s[0][0][0]);
          glEvalMesh2(type, 0, grid, 0, grid);
          
          numVertices += grid * grid;
        } // if
      } // for i

      // Draw the tiny quads at the top and bottom.
      
      glBegin(GL_QUADS);
      
        glTexCoord2f(.025f,.025f);

        glNormal3f(0, 0, 1);
        glVertex3f(0,.03f,cpdata[127][2]);
        glVertex3f(.03f,0,cpdata[127][2]);
        glVertex3f(0,-.03f,cpdata[127][2]);
        glVertex3f(-.03f,0,cpdata[127][2]);

        glNormal3f(0, 0, -1);
        glVertex3f(-.03f,0,cpdata[131][2]);
        glVertex3f(0,-.03f,cpdata[131][2]);
        glVertex3f(.03f,0,cpdata[131][2]);
        glVertex3f(0,.03f,cpdata[131][2]);

        // Close off the spout.

        glNormal3f(0, 0, 1);

        for (i = 135; i < 143; ++i)
        {
          glVertex3fv(cpdata[i]);
        } // for

      glEnd();

    glPopMatrix();
    
  glPopAttrib();
  
  return numVertices;
} // DrawTeapot
