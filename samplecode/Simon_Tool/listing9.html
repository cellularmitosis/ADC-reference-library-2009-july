  integer        version = kThisToolVersion;

  byte        persistentOff = 0x0000, 
            persistentOn = 0x0001;
  byte        dontContinue = 0x0000, 
            continue = 0x0001;
};

              kDefaultType Resource Information
************************************************************************************/

  kThisToolVersion,

  continue
