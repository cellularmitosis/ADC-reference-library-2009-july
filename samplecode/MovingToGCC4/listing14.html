template <class T1> struct A {
  template <int i> static void func1 ();
};

/* GCC 4.0 requires proper use of the 'template' keyword.  In this case,
   the 'BAD' version has '<' referring to the less-than operator, which
   is not what is desired.  See ISO C++, [temp.names] paragraph 4.  */

template <class T2> void func2() {
#if BAD
  A<T2>::func1<0>();
#else
  A<T2>::template func1<0>();
#endif
};
