#define SwapS32Multiple(ptr,cnt) \
{   long *__p = (long *)(ptr); \
  long __i; \
  for (__i=0;__i<(long)(cnt);__i++) {\
    SwapS32(*__p); \
    __p++; \
  } \
}
#define SwapU32Multiple(ptr,cnt) \
{   unsigned long *__p = (unsigned long *)(ptr); \
  long __i; \
  for (__i=0;__i<(long)(cnt);__i++) {\
    SwapU32(*__p); \
    __p++; \
  } \
}
