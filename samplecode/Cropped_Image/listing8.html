#import "FinderStyleCropMarker.h"

@implementation FinderStyleCropMarker

- (void) drawCropMarker
  { 
  if (!NSIsEmptyRect(selectedRect))
    { 
    [fillColor set]; 
    NSRectFillUsingOperation(selectedRect, NSCompositeSourceOver); 
    [strokeColor set]; 
    NSFrameRect(selectedRect);
    }
  }
  
@end
