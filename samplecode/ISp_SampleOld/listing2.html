{
  currentVersion,
  {
    "Default (Keyboard)", 0, kISpDeviceClass_Keyboard, 
      kISpKeyboardID_Apple, notApplSet, isDefaultSet, 
      ktset_DefaultKeyboard,

      kISpKeyboardID_Apple, isApplSet, notDefaultSet, 
      ktset_LeftyKeyboard,

/* Default keyboard set */

{
  supportedVersion,
  {  
    /* kNeed_FireWeapon */
    kpd0Key,

    kpd9Key,
     rNoModifiers,

     rNoModifiers,
     
    kpd6Key,         /* max (right/up/forward) */
     rNoModifiers,

     rNoModifiers,
     
    kpd8Key,         /* max (right/up/forward) */
     rNoModifiers,
    
    /* kNeed_Yaw */
    /* this need does not generate any items, because it has later button equivalents */

    /* kNeed_Throttle */
    /* this need does not generate any items, because it has later button equivalents */
  
  
    /* kNeed_StartPause */
    tildeKey,
     rNoModifiers,
    
    /* kNeed_Quit */
    qKey,
     rControlOff, rOptionOff, rShiftOff, controlOff, optionOff, shiftOff, commandOn,
    
    /* kNeed_Weapon_MachineGun */
    n1Key,
     rNoModifiers,
    /* kNeed_Weapon_Cannon */
    n2Key,
     rNoModifiers,
    /* kNeed_Weapon_Laser */

     rNoModifiers,
    /* kNeed_Weapon_Missle */

     rNoModifiers,
    /* kNeed_Weapon_PrecisionBomb */
    n5Key,
     rNoModifiers,
    /* kNeed_Weapon_ClusterBomb */
    n6Key,
     rNoModifiers,
  

    /* kNeed_Roll_AsDelta */
    /* this need does not generate any items \xD0 the keyboard does not do deltas  */

    /* kNeed_Pitch_AsDelta */
    /* this need does not generate any items \xD0 the keyboard does not do deltas  */

    /* kNeed_Yaw_AsDelta */
    /* this need does not generate any items \xD0 the keyboard does not do deltas  */

    aKey,
     rNoModifiers,
    /* kNeed_Yaw_Center */
    sKey,
     rNoModifiers,
    /* kNeed_Yaw_Right */
    dKey,
     rNoModifiers,

     rNoModifiers,

  };
