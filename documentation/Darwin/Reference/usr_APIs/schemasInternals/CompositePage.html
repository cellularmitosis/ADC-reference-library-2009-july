    const xmlChar *version; 
    const xmlChar *id; 
    xmlDocPtr doc; 
    xmlSchemaAnnotPtr annot; 
    int flags;  
    xmlHashTablePtr typeDecl; 
    xmlHashTablePtr attrDecl; 
    xmlHashTablePtr attrgrpDecl; 
    xmlHashTablePtr elemDecl; 
    xmlHashTablePtr notaDecl;
