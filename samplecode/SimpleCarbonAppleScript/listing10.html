#import <Foundation/Foundation.h>

#import "Scriptability.h"
#import "ScriptingCommands.h"
#import "scriptLog.h"


int InitializeScriptability() {
  OSStatus err;
  NSAutoreleasePool* pool;
  scriptingCommands* myCommands = nil;
  
    /* default result */
  err = memFullErr;
  
  scriptLog(@"Scriptability InitializeScriptability");
  
    /* auto release pool around Cocoa calls from Carbon */
  pool = [[NSAutoreleasePool alloc] init];
    
    /* create our object */
  myCommands = [[scriptingCommands alloc] init];
  if ( myCommands != nil ) {
  
    scriptLog(@"Scriptability InitializeScriptability ( myCommands != nil )");
    
      /* set the top level object for the Cocoa scripting runtime */
    [[NSScriptExecutionContext sharedScriptExecutionContext] setTopLevelObject:myCommands];
      
      /* build the script suite registry from our .sdef file  */
    [NSScriptSuiteRegistry sharedScriptSuiteRegistry];
    
      /* success */
    err = noErr;
  }
  
    /* done with the autorelease pool */
  [pool release];
  
  scriptLog(@"Scriptability complete");
  return err;
}
