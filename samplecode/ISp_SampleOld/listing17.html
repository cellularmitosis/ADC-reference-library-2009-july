void
main(void)
{
  ToolboxInit();
  AppleEventsInit();

  RegisterAppearanceClient();
  
  MenuInit();
  EventInit();

  EventLoop();

  AppleEventsShutDown();

  UnregisterAppearanceClient();

  ExitToShell();
}

//*  Ñ\xD1\xD1\xD1\xD1  ToolboxInit

static void
ToolboxInit(void)
{
  MaxApplZone();

  InitGraf(&qd.thePort);
  InitFonts();
  InitWindows();
  InitMenus();
  TEInit();
  InitDialogs(nil);
  InitCursor();
}
