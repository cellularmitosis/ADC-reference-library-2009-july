#ifndef __RECORD_SOUND__

//So we can pass our A5 and other info to our PBWrite completion routines
typedef struct myRecordingParamBlockRec {
  ParamBlockRec      pb;
  long          myA5;
  Boolean          pbInUse,
              lastWrite;
} myRecordingParamBlockRec, *myRecordingParmBlkPtr;

//Keep track of the info needed to record
typedef struct {
  long          sanitycheck;

  myRecordingParmBlkPtr  pb0,
              pb1;
  Ptr            recBuffer0,
              recBuffer1,
              resampledBuf0,
              resampledBuf1;
  SoundConverter      sc;
  SoundComponentData    inputFormat,
              outputFormat;
  Fixed          sampleRate;
  OSType          compression;
  unsigned long      totalBytes,
              inputFrames;
  long          myA5,
              devBuffer,
              parID,
              soundRefNum,
              initialOffset;
  short          whichBuffer,
              fileRefNum,
              volRefNum,
              numChannels,
              sampleSize,
              AGC;
  Str255          nameString;
} Vars, *VarsPtr;
