static void Message_Concatenate(char *typeStr, long value, char *message, char *outputMessage)
{
  char  *slash, *period, *start;

  start = message;

  /* Find first period in file name at end of path name */
  period = strchr(message, '.');
  if (period != NULL) {
    *period = '\0';

    /* Find last slash in path name */
    slash = strrchr(message, '/');
    if (slash != NULL) {
      /* No path name */
      start = slash+1;
    }

    *period = '.';
  }

  sprintf (outputMessage, "QD3D %s %ld: %s", typeStr, value, start);
}

static void Message_SecretError(TQ3Error error, char *message)
{
  char  outputMessage[512];

  Message_Concatenate("Error", error, message, outputMessage);

#if defined(OS_MACINTOSH) && OS_MACINTOSH
  debugstr(outputMessage);
#endif
}
