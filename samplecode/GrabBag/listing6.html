#include <Carbon.h>

#include  "Main.h"

static  void  CenterCursorWindowOnPoint( WindowRef window, Point *mouseLocation );
static  void  DrawCursorWindowContents ( Point* mouseLocation );


extern  GlobalsStruct  g;

//  CursorGrande additions
//  When the mouse moves, update the cursor window position and redraw the contents
pascal  OSStatus  LargeCursorEventHandler( EventHandlerCallRef myHandler, EventRef event, void* userData )
{
  #pragma unused ( myHandler )
    static  Point    mouseLocation;
    OSStatus      result      = eventNotHandledErr;
  UInt32        eventKind    = GetEventKind( event );
  WindowinfoStruct  *windowInfo;//    = (WindowRef) userData;
    
    if ( eventKind == kEventWindowBoundsChanging )
    {
      Rect    oRect, newRect;
      WindowRef  window;
      GetEventParameter( event, kEventParamDirectObject, typeWindowRef, NULL, sizeof(WindowRef), NULL, &window );
      windowInfo  = (WindowinfoStruct*) GetWRefCon( window );
      if ( (windowInfo==NULL) || ( (windowInfo->windowidentifier!=kMainTabWindowType) && (windowInfo->windowidentifier!=kThreadWindowType) ) )
        return( eventNotHandledErr );
      GetEventParameter( event, kEventParamPreviousBounds, typeQDRectangle, NULL, sizeof(Rect), NULL, &oRect );
      GetEventParameter( event, kEventParamCurrentBounds, typeQDRectangle, NULL, sizeof(Rect), NULL, &newRect );
    mouseLocation.h  += newRect.left - oRect.left;
    mouseLocation.v  += newRect.top - oRect.top;
    }
    else
    {
        //  Get the mouse location
      GetEventParameter( event, kEventParamMouseLocation, typeQDPoint, NULL, sizeof(mouseLocation), NULL, &mouseLocation );
    }
    
    CenterCursorWindowOnPoint( g.cursorWindow, &mouseLocation );
    DrawCursorWindowContents( &mouseLocation );

    return( result );
}

//  Given a point in global coordinates, center the cursor window over that location
static  void  CenterCursorWindowOnPoint( WindowRef window, Point *mouseLocation )
{
    Rect    bounds;
    Point    windowLocation;
    
    GetPortBounds( GetWindowPort(window), &bounds );
    
    windowLocation.h  = mouseLocation->h - (bounds.right - bounds.left) / 2;
    windowLocation.v  = mouseLocation->v - (bounds.bottom - bounds.top) / 2;
    
    //  We don't need to transform global<->local coords since MoveWindow takes global coords
    MoveWindow( window, windowLocation.h, windowLocation.v, 1 );
}

static  void  DrawCursorWindowContents( Point* mouseLocation )
{
    CGrafPtr    oPort;
    Rect      bounds;
    RGBColor    rgbRedColor    = { 0xFFFF, 0, 0 };
    RGBColor    rgbGreenColor  = { 0, 0xFFFF, 0 };
    RGBColor    rgbBlueColor  = { 0, 0, 0xFFFF };

    GetPort( &oPort );
    GetWindowPortBounds( g.cursorWindow, &bounds );
    SetPortWindowPort( g.cursorWindow );
    
  PenSize( 6, 6 );
  RGBForeColor( &rgbBlueColor);  FrameOval( &bounds );  InsetRect( &bounds, 9, 9 );
  RGBForeColor( &rgbRedColor);  FrameOval( &bounds );  InsetRect( &bounds, 9, 9 );
  RGBForeColor( &rgbGreenColor);  FrameOval( &bounds );

    SetPort( oPort );
}
