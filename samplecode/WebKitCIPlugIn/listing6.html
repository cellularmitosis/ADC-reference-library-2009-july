<script LANGUAGE="JavaScript">
function brighter() 
{
  var plugin = document.embeds["CoreImagePlugin"];
  plugin.gammaValue = plugin.gammaValue + 0.1;
  plugin.setGamma();
}

function darker() 
{
  var plugin = document.embeds["CoreImagePlugin"];
  plugin.gammaValue = plugin.gammaValue - 0.1;
  plugin.setGamma();
} 

function setImage() 
{
  var plugin = document.embeds["CoreImagePlugin"];
  plugin.imageURL = document.webkitTest.urlField.value;
  plugin.setImage();
} 
</script>
