/*  Convenience macros: shortcuts for doing common tasks  */

/*
  Use the following three macros to produce entry/exit traces. Usage is:
  
  void  foo()
  {
    TRACE_SETUP;
    LOG_ENTRY( "MyApp:foo" );
  
    // foo's code ...
    
    LOG_EXIT;
  }
*/

#define  TRACE_SETUP    static InstTraceClassRef __sTrace = 0; InstEventTag __tTag

#define  LOG_ENTRY( n )  if ( INSTRUMENTATION_ACTIVE && ( __sTrace || noErr == InstCreateTraceClass( kInstRootClassRef, n, 0, kInstEnableClassMask, &__sTrace))) \
            { __tTag = InstCreateEventTag(); InstLogTraceEvent( __sTrace, __tTag, kInstStartEvent); }

#define  LOG_SUMM_ENTRY( n )  if ( INSTRUMENTATION_ACTIVE && ( __sTrace || noErr == InstCreateTraceClass( kInstRootClassRef, n, 0, kInstEnableClassMask | kInstSummaryTraceClassMask, &__sTrace))) \
              { __tTag = InstCreateEventTag(); InstLogTraceEvent( __sTrace, __tTag, kInstStartEvent); }

#define  LOG_EXIT  if ( INSTRUMENTATION_ACTIVE)  InstLogTraceEvent( __sTrace, __tTag, kInstEndEvent)


#endif /* __INSTRUMENTATIONMACROS__ */
