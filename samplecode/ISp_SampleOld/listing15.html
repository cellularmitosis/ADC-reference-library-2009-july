struct Input_GameState
{
  Boolean    gameInProgress;
  Boolean    gamePaused;

  Boolean    fireWeaponState;
  UInt32    fireWeaponCount;
  
  SInt16    currentWeapon;
  
  SInt32    rollInput;
  SInt32    pitchInput;
  SInt32    yawInput;
  SInt32    throttleInput;
  
  Fixed    deltaRoll;
  Fixed    deltaPitch;
  Fixed    deltaYaw;
};
typedef struct Input_GameState Input_GameState;
