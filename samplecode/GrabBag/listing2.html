void  CreateMyDataBrowserControl( WindowRef window )
{
  const  ControlID      dbControlID      = { 'Brow', 0 };
  OSStatus          err;
  ControlRef          dbControl;
  DataBrowserCustomCallbacks  dbCustomCallbacks;
  DataBrowserItemID      dataBrowserItemID  = 1;            //  This will be our "row" ID
  
  GetControlByID( window, &dbControlID, &dbControl );
  
  dbCustomCallbacks.version          = kDataBrowserLatestCallbacks;
   InitDataBrowserCustomCallbacks( &dbCustomCallbacks );
  dbCustomCallbacks.u.v1.drawItemCallback    = NewDataBrowserDrawItemUPP( DBCustomDrawProc );  //  Need a custom draw proc for colored text
  SetDataBrowserCustomCallbacks( dbControl, &dbCustomCallbacks );
  
  err  = AddDataBrowserItems( dbControl, kDataBrowserNoItem, 1, &dataBrowserItemID, kDataBrowserItemNoProperty );    //  Create row "1"
  dataBrowserItemID++;
  err  = AddDataBrowserItems( dbControl, kDataBrowserNoItem, 1, &dataBrowserItemID, kDataBrowserItemNoProperty );    //  Create row "2"
  
  SetDataBrowserSelectionFlags( dbControl, kDataBrowserCmdTogglesSelection );
  
  SetAutomaticControlDragTrackingEnabledForWindow( window, true );      //  Needed to automatically support moving columns
  
  return;
}

static pascal void DBCustomDrawProc( ControlRef browser, DataBrowserItemID item, DataBrowserPropertyID property, DataBrowserItemState itemState, const Rect *theRect, SInt16 gdDepth, Boolean colorDevice)
{
  RGBColor      color;
  CFStringRef      cfString;
  long        systemVersion;

  cfString  = CFStringCreateWithFormat( NULL, NULL, CFSTR("Row %d"), item );
  
  if ( itemState == kDataBrowserItemIsSelected )      //  In this sample we handle the "selected" state, all others fall through to our "active" state
  {
    Gestalt( gestaltSystemVersion, &systemVersion );
    if ( (systemVersion >= 0x00001030) && (IsControlActive( browser ) == false) )  //  Panther DB starts using kThemeBrushSecondaryHighlightColor for inactive browser hilighting
      SetThemePen( kThemeBrushSecondaryHighlightColor, 32, true );
    else
      SetThemePen( kThemeBrushPrimaryHighlightColor, 32, true );

    PaintRect( theRect );                //  First paint the hilite rect, then the text on top
  }

  BlockZero( &color, sizeof(color) );            //  set the color for our text
  if ( property == 'Red ' )    color.red  = 0xFFFF;
  else if ( property == 'Blue' )  color.blue  = 0xFFFF;
  else              color.green  = 0xFFFF;

#if ( (0) && (__APPLE_CC__) )  //  This section shows how to draw colored text using CG.  (Available directly through Mach-O, hence the __APPLE_CC__ for this sample)
  {
    CGContextRef    cgContext;
    GrafPtr        port;
    GetPort( &port );
    QDBeginCGContext( port, &cgContext );
    CGContextSetRGBFillColor( cgContext, (float)color.red / (float)65535, (float)color.green / (float)65535, (float)color.blue / (float)65535, 1.0 );
    DrawThemeTextBox( cfString, kThemeWindowTitleFont, kThemeStateActive, true, theRect, teFlushDefault, cgContext );
    QDEndCGContext( port, &cgContext );
  }
#else
  {
  RGBColor      oColor;
    GetForeColor( &oColor );
    RGBForeColor( &color );
    DrawThemeTextBox( cfString, kThemeApplicationFont, kThemeStateActive, true, theRect, teFlushDefault, NULL );  //  Draw anti-aliased colored text
    RGBForeColor( &oColor );
  }
#endif
  if ( cfString != NULL )  CFRelease( cfString );
}
